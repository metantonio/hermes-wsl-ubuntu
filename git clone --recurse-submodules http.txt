# This downloads dflash and llama.cpp fork also
LUCEBOX_HUB_DIR="$REAL_HOME/lucebox"
mkdir LUCEBOX_HUB_DIR
git clone --recurse-submodules https://github.com/Luce-Org/lucebox-hub LUCEBOX_HUB_DIR
cd LUCEBOX_HUB_DIR
cd dflash/

# Now we are gonna build llama.cpp for qwen3.6:27b with D-FLASH, carefull with DCMAKE_CUDA_ARCHITECTURES because it depends of the GPU (e.g A6000 is 86), verify architecture first:

python -c "import torch; p=torch.cuda.get_device_properties(0); print(p.name, 'sm_%d%d'%(p.major,p.minor), p.multi_processor_count,'SMs', round(p.total_memory/1e9,1),'GB')"
nvcc --version

# Build llama.cpp:
cmake -B build -S . -DCMAKE_BUILD_TYPE=Release -DCMAKE_CUDA_ARCHITECTURES=86 -DDFLASH27B_ENABLE_BSA=ON

cmake --build build --target test_dflash test_flashprefill_kernel -j
